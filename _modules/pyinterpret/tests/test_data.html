import unittest

import numpy as np

from pyinterpret.data.dataset import DataSet

[docs]    def setUp(self):
        """Create data for testing"""
        self.parser = create_parser()
        args = self.parser.parse_args()
        debug = args.debug

        self.dim = 3
        self.n = 8

        self.X = np.array([
            [0, 0, 0],
            [0, 0, 2],
            [0, 1, 0],
            [0, 2, 1],
            [1, 0, 0],
            [1, 0, 1],
            [1, 1, 0],
            [2, 1, 1],
        ])

        self.feature_names = ['x{}'.format(i) for i in range(self.dim)]
        self.index = ['{}'.format(i) for i in range(self.n)]

        if debug:
            self.log_level = 10
        else:
            self.log_level = 30



[docs]    def test_dataset_data_returns_original(self):
        """
        ensure Dataset(data).data == data
        """
        data_set = DataSet(self.X, feature_names=self.feature_names, index=self.index)
        assert_array_equal(data_set.data, self.X)



[docs]    def test_1D_numpy_array(self):
        """
        ensure Dataset(data) works when data is 1D np.ndarray
        """

        feature_id = 0
        array_1d = self.X[:, feature_id][:, np.newaxis]
        feature_names = [self.feature_names[feature_id]]

        data_set = DataSet(array_1d,
                           feature_names=feature_names,
                           index=self.index,
                           log_level=self.log_level)
        assert_array_equal(data_set.data, array_1d)
        assert data_set.feature_ids == feature_names

        """

                           index=self.index,
                           log_level=self.log_level)
        assert_array_equal(data_set.data, array_2d)
        assert data_set.feature_ids == feature_names

        """
        Ensure Dataset(data) works when data is pd.DataFrame
        """
        X_as_dataframe = pd.DataFrame(self.X, columns=self.feature_names, index=self.index)
        data_set = DataSet(X_as_dataframe, log_level=self.log_level)

        feature_names_warning = "Feature Names from DataFrame not loaded properly"

        assert data_set.feature_ids == self.feature_names
        assert data_set.index == self.index
        assert_array_equal(data_set.data, self.X)


[docs]    def test_generate_grid_1_variable(self):
