"Specialized Exceptions for pyinterpret"


[docs]def exception_factory(exception_name, base_exception = Exception, attributes = None):
    attribute_dict = {
        "__init__":base_exception.__init__
    }
    if isinstance(attributes, dict):
        attributes.update(attributes)
    return type(
        exception_name,
        (base_exception, ),
        attribute_dict
    )


DataSetNotLoadedError = exception_factory('DataSetNotLoadedError')
